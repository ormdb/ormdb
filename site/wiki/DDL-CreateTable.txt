<k>CreateTable</k> function is used to create a new table in a database.

<br/>

<st>Definition</st>
<code-tab>
	<code lang="c#">
		namespace Connection {
			public class DDL {

			   public List<string> CreateTables(IEnumerable<Table.Base> Tables,
										string SchemaPrefix = Manager.DefaultSchemaPrefix,
										bool CreateIndexes = true,
										bool OnlyGenerateScript = false){}

			   public List<string> CreateTable(Table.Base Table,
										string SchemaPrefix = Manager.DefaultSchemaPrefix,
										bool CreateIndexes = true,
										bool OnlyGenerateScript = false){}

			   public string CreateTable(string TableName,
										IEnumerable<Table.Column> Columns, 
										IndexInfo PrimaryKeyInfo = null, 
										string SchemaPrefix = Manager.DefaultSchemaPrefix, 
										bool OnlyGenerateScript = false){}
	</code>
	<code lang="vb">
		NameSpace Connection
			Public Class DDL
	
			   Public Function CreateTables(Tables As IEnumerable(Of Table.Base),
											SchemaPrefix As String = Manager.DefaultSchemaPrefix,
											CreateIndexes As Boolean = True,
											OnlyGenerateScript As Boolean = False) As List(Of String)

			   Public Function CreateTable(Table As Table.Base,
										   SchemaPrefix As String = Manager.DefaultSchemaPrefix,
										   CreateIndexes As Boolean = True,
										   OnlyGenerateScript As Boolean = False) As List(Of String)

			   Public Function CreateTable(TableName As String,
										   Columns As IEnumerable(Of Table.Column), 
										   PrimaryKeyInfo As IndexInfo = Nothing, 
										   SchemaPrefix As String = Manager.DefaultSchemaPrefix, 
										   OnlyGenerateScript As Boolean = False) As String
	</code>
</code-tab>
<co type="light">
	<k>SchemaPrefix</k> is a prefix for table-name. (Exm. Name="table-name" , NameWithPrefix="[pfx]table-name").
</co>

<br/>

<st>Examples</st>

<ew/>

<br/>

Following section demonstrates creating a table.

<code-tab>
	<code lang="c#">
		// create the table.
		cn.DDL.GetTableScript(h);
	</code>
	<code lang="vb">
		' create the table.
		cn.DDL.GetTableScript(h)
	</code>
</code-tab>

<br/>

Following section demonstrates how to generate <k>CREATE TABLE</k> script.

<code-tab>
	<code lang="c#">
		var script1 = cn.DDL.GetTableScript(h, OnlyGenerateScript:true);
	</code>
	<code lang="vb">
		Dim script1 = cn.DDL.GetTableScript(h, OnlyGenerateScript:=True)
	</code>
</code-tab>
