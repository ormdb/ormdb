Examples 

			//---------------------------------------------------
			// instance(s) created from definition(s)
			var t1 = new TableHeader();

			// new instance of a table from database
			var t2 = cn.DDL.GetTableInstance();


			//---------------------------------------------------
			// Serialization

			string serialize_data = "";

			serialize_data = Table.Base.Serialize(t1);
			serialize_data = Table.Base.Serialize(t2);


			//---------------------------------------------------
			// Deserialization
			var new_instance = Table.Base.Deserialize(serialize_data);
