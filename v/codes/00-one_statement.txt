var th = new TableHeader();
var tl = new TableLines();
	
var sql = Odb.SQL.Select(x => 
				x.From(th)
				.LeftJoin(tl, tl.Year == th.Year & 
								tl.Id == th.Id ) 
				.Where( th.Year > 2018 )
			);	

... sql.ExecuteReader(connection_for_mssql);
... sql.ExecuteReader(connection_for_oracle);
... sql.ExecuteReader(connection_for_mysql);
... sql.ExecuteReader(connection_postgres);
