		var t0 = new Orders();
		var t1 = new OrderDetails();
		
		Odb.SQL.Select(x => 
			    x.From(t0)
			     .Fields(t0.OrderID, t1.OrderDetailID)
			     .LeftJoin(t1, t1.OrderID == t0.OrderID) 
			     .Where( t0.OrderID > 1500 &
					     t1.OrderID != null )
		);
SELECT t0.OrderID, t1.OrderDetailID
FROM [Orders] AS t0
LEFT JOIN [OrderDetails] AS t1
ON (t1.OrderID = t0.OrderID)
WHERE ((t0.OrderID > 1500) AND (t1.OrderID IS NOT NULL))
