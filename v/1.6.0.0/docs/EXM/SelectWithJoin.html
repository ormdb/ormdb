SELECT t0.*, t1.OrderDetailID, t1.ProductID, t1.Quantity
FROM [Orders] AS t0
LEFT JOIN [OrderDetails] AS t1
ON (t1.OrderID = t0.OrderID)
WHERE ((t0.OrderID > 1500) AND (t1.OrderID IS NOT NULL))
