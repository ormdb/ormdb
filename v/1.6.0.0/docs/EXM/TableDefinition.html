Following section demonstrates how to define the tables. 
using System;
using System.Linq.Expressions;
using Odb.Table;

// Attributes are defined at once to use with [Ref]...
public enum Ref
{
    [Column("year", Length = 4, DefaultValue = 2000, MinValue = 1900, MaxValue = 2100)] year,
    [Column("note", Length = 100, DefaultValue = "")] note,
    [Column("id", Length = 2)] id,
    [Column(DateFormat = ColumnDateFormat.OnlyDate)] only_date
}

[Table("header")]
public class TableHeader : Base
{
    [Column(Ref.year)]                public int? Year;
    [Column(Ref.id)]                  public int? Id;
    [Column("date", Ref.only_date)]   public DateTime? DocDate;
    [Column("active")]                public bool IsActive;
    [Column(Ref.note)]                public string Note;

    // Method expression which returns of type "TableConfig", is returned.
    protected override Expression<TableConfig> Config() =>
        (x) => x.SetPrimaryKey(Year, Id)
                        .AddIndex(DocDate);
}

[Table("lines")]
public class TableLines : Base
{
    [Column(Ref.year)]                public int? Year;
    [Column(Ref.id)]                  public int? Id;
    [Column("lineid", Length = 2)]    public int? LineId;
    [Column(Ref.note)]                public string Note;

    // Method expression which returns of type "TableConfig", is returned.
    protected override Expression<TableConfig> Config() =>
        (x) => x.SetPrimaryKey(Year, Id, LineId);
}

Imports System
Imports Odb.Table
Imports System.Linq.Expressions

' column attributes are defined at once to use with [Ref]...
Public Enum Ref
    <Column("year", Length:=4, DefaultValue:=2000, MinValue:=1900, MaxValue:=2100)> year
    <Column("note", Length:=100)> note
    <Column("id", Length:=2)> id
    <Column(DateFormat:=ColumnDateFormat.OnlyDate)> only_date
End Enum

<Table("header")>
Public Class TableHeader
    Inherits Base

    <Column(Ref.year)>              Public Year As Integer?
    <Column(Ref.id)>                Public Id As Integer?
    <Column("date", Ref.only_date)> Public DocDate As Date?
    <Column("active")>              Public IsActive As Boolean?
    <Column(Ref.note)>              Public Note As String

    ' Method expression which returns of type "TableConfig", is returned.
    Protected Overrides Function Config() As Expression(Of TableConfig)
        Return Function(x) x.SetPrimaryKey(Year, Id) _
                            .AddIndex(DocDate)
    End Function
End Class

<Table("lines")>
Public Class TableLines
    Inherits Base

    <Column(Ref.year)>              Public Year As Integer?
    <Column(Ref.id)>                Public Id As Integer?
    <Column("lineid", Length:=2)>   Public LineId As Short?
    <Column(Ref.note)>              Public Note As String

    ' Method expression which returns of type "TableConfig", is returned.
    Protected Overrides Function Config() As Expression(Of TableConfig)
        Return Function(x) x.SetPrimaryKey(Year, Id, LineId)
    End Function
End Class
