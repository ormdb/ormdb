Following section demonstrates to use Set method for setting value(s). 
		var th = new TableHeader();
		
		var sql = Odb.SQL.Update(x =>
						x.From(th)
						 .Where(th.Year == 2018)
						 .Set(th.DocDate, System.DateTime.Now)
						 .SetRaw(th.Id, th.Id + 1)
						 .SetRaw(th.IsActive, !th.IsActive)
						 .SetNull(th.Note)
					);
	
		Dim th = new TableHeader()
		
		Dim sql = Odb.SQL.Update(Function(x) _
						x.From(th) _
						 .Where(th.Year = 2018) _
						 .Set(th.DocDate, System.DateTime.Now) _
						 .SetRaw(th.Id, th.Id + 1) _
						 .SetRaw(th.IsActive, Not th.IsActive) _
						 .SetNull(th.Note)
					)
UPDATE th

FROM [header] AS th
