        var th = new TableHeader();

        Odb.SQL.Modify<TableHeader>(x => 
                                x.From(th)
                                 .Where(th.Year == 2021 & 
                                        th.Id == 1000)
        ).onInsert(Row => {
                        Row.DocDate = DateTime.Now
                    }
        ).onUpdate((Row, OldRow) => {
                        Row.Note = "new note value"
                    }
        ).onAlways((Row, OldRow) => {
                        var IsUpdate = (OldRow is object);
                    }
        ).ExecuteNonQuery(cn);
    
        Dim th = new TableHeader()

        Odb.SQL.Modify(Of TableHeader)(Function(x) _
                                x.From(th) _
                                 .Where(th.Year = 2021 And
                                        th.Id = 1000)
        ).onInsert(Sub(Row)
                        Row.DocDate = DateTime.Now
                    End Sub) _
         .onUpdate(Sub(Row, OldRow)
                        Row.Note = "new note value"
                    End Sub) _
         .onAlways(Sub(Row, OldRow)
                        Dim IsUpdate = (OldRow IsNot Nothing)
                    End Sub) _
        .ExecuteNonQuery(cn)
