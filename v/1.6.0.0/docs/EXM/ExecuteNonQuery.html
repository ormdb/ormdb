Following section demonstrates how to execute the INonQueryable expression. 
		var th = new TableHeader();
		
		var sql = Odb.SQL.Delete(x => x.From(th).Where(th.Id == 12345M) );

		var affectedRows = sql.ExecuteNonQuery(cn);
	
		Dim th = new TableHeader()
		
		Dim sql = Odb.SQL.Delete(Function(x) x.From(th).Where(th.Id = 12345D) )
		
		Dim affectedRows = sql.ExecuteNonQuery(cn)


Following section demonstrates how to execute the TSQL command. 
		// SQL statement is being prepared
		var str = "DELETE FROM table WHERE field1<>0 AND field2='X'";

		var affectedRows = cn.ExecuteNonQuery(str);
	
		' SQL statement is being prepared
		Dim str = "DELETE FROM table WHERE field1<>0 AND field2='X'"

		Dim affectedRows = cn.ExecuteNonQuery(str)


Following section demonstrates how to execute the DbCommand. 
		// DbCommand is being prepared
		var cmd = cn.CreateCommand(
					"DELETE FROM table WHERE field1<>@p1 AND field2=@p2",
					new { p1 = 0, p2 = "X"});
		
		var affectedRows = sql.ExecuteNonQuery(cmd);
	
		' DbCommand is being prepared
		Dim cmd = cn.CreateCommand(
					"DELETE FROM table WHERE field1<>@p1 AND field2=@p2",
					new { p1 = 0, p2 = "X"})
		
		Dim affectedRows = sql.ExecuteNonQuery(cmd)
