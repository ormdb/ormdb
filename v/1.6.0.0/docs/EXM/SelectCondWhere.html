		var sql1 = Odb.SQL.Select(x =>

							 .Where( th.Year == 2017 ));
		if ( true )
			sql1.Add(x => x.Where(th.IsActive == true));
		else
			sql1.Add(x => x.Where(th.IsActive == false));


		// [Where] clause is being executed depending on value test.
		var val1 = "X";
		var sql2 = Odb.SQL.Select(x => 
							x.From(th)
							 .Where( (val1 == "X" & th.IsActive == true) |

									)
					);
