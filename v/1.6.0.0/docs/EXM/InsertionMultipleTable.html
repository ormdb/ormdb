		var th = new TableHeader();
		var tl = new TableLines();

		var sql = Odb.SQL.Insert(true);

		for (short n = 0; n < 100; n++)
		{
			var row1 = th.NewModel();
				row1.Year    = DateTime.Now.Year;
				row1.Id      = n;
				row1.DocDate = DateTime.Now.AddDays(1);
				row1.Note    = "header note value";
			sql.Add(row1);

			for (short i = 0; i < 2; n++)
			{
				var row2 = tl.NewModel();
					row2.Year   = row1.Year;
					row2.Id     = row1.Id;
					row2.LineId = i;
					row2.Note   = "line note value";
				sql.Add(row2);
			}
		}
