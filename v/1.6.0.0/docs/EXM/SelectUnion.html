		var th = new TableHeader();
		var tl = new TableLines();
		
		var sql1 = Odb.SQL.Select(x =>
						x.From(th)
						 .GroupBy(th.Year)
						 .OrderDESC(th.DocDate)
						 .Fields(th.Year, x[th.Id].MAX().AS("C1"))
					);
		var sql2 = Odb.SQL.Select(x =>
						x.From(tl)
						.GroupBy(tl.Year)
						.Fields(tl.Year, x[th.Id].MAX().AS("C1"))
					);

		var sql = Odb.SQL.Union(x =>
						x.RemoveDuplicates()
						 .Add(sql1)
						 .Add(sql2)
						 .OrderASC(sql1[th.Year]).OrderDESC(sql1["C1"])
					);
	
		Dim th = new TableHeader()
		Dim tl = new TableLines()
		
		Dim sql1 = Odb.SQL.Select(Function(x) _
						x.From(th) _
						 .GroupBy(th.Year) _
						 .OrderDESC(th.DocDate) _
						 .Fields(th.Year, x(th.Id).MAX().AS("C1"))
					)
		Dim sql2 = Odb.SQL.Select(Function(x) _
						x.From(tl) _
						.GroupBy(tl.Year) _
						.Fields(tl.Year, x(tl.Id).MAX().AS("C1"))
					)

		Dim sql = Odb.SQL.Union(Function(x) _
						x.RemoveDuplicates() _
						 .Add(sql1) _
						 .Add(sql2) _
						 .OrderASC(sql1(th.Year)).OrderDESC(sql1("C1"))
					)
SELECT * FROM (
(
SELECT t3.year, MAX(t3.id) AS C1
FROM [header] AS t3
GROUP BY t3.year
ORDER BY t3.date DESC
) AS t0
UNION
(
SELECT t1.year, MAX(t1.id) AS C1
FROM [lines] AS t1
GROUP BY t1.year
) AS t2
) AS t
ORDER BY year ASC, C1 DESC
