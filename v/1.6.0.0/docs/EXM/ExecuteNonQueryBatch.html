Following section demonstrates how to execute multiple commands in one batch-query with ExecuteNonQuery. 
		var th = new TableHeader();
		var tl = new TableLines();
		
		var sql = SQL.X.Query();
		
		// DELETE commands are being queued...
		sql.Delete(th);
		sql.Delete(tl);
		
		// INSERT commands are being queued...
		sql.Insert(True).Add(new TableHeader() {
                                Year = DateTime.Now.Year,
								Id = 1,
								DocDate = DateTime.Now.AddDays(1)
                            })

		// UPDATE commands are being queued...
		sql.Update(x =>
				x.From(th)
				.Where(th.Year == 2018)
				.Set(th.IsActive, !th.IsActive)
			);
		
		// Commands are being executed...
		sql.ExecuteNonQuery(cn);
	
		Dim th = New TableHeader()
		Dim tl = New TableLines()
		
		Dim sql = SQL.X.Query()
		
		' DELETE commands are being queued...
		sql.Delete(th)
		sql.Delete(tl)
		
		' INSERT commands are being queued...
		sql.Insert(True).Add(New TableHeader() With {
                                .Year = DateTime.Now.Year,
								.Id = 1,
								.DocDate = DateTime.Now.AddDays(1)
                            })

		' UPDATE commands are being queued...
		sql.Update(Function(x) _
				x.From(th) _
				.Where(th.Year = 2018) _
				.Set(th.IsActive, Not th.IsActive)
			)
		
		' Commands are being executed...
		sql.ExecuteNonQuery(cn)
