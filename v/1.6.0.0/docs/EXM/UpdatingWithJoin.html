Following section demonstrates how to update from [TableHeader] records, which contain records that is between 1-10 in [TableLines]. 
		var th = new TableHeader();
		var tl = new TableLines();
		
		var sql = Odb.SQL.Update(x =>
						x.From(th) 
						 .InnerJoin(tl, tl.Year == th.Year & tl.Id == th.Id) 
						 .Where(tl.LineId > 1M &
								tl.LineId < 10M)
						 .Set(th.IsActive, true)
					);

		Dim tl = new TableLines()
		
		Dim sql = Odb.SQL.Update(Function(x) _
						x.From(th) _
						 .InnerJoin(tl, tl.Year = th.Year And tl.Id = th.Id) _
						 .Where(tl.LineId > 1D And
								tl.LineId < 10D) _
						 .Set(th.IsActive, True)
					)

		Dim affectedRows = sql.ExecuteNonQuery(cn)
