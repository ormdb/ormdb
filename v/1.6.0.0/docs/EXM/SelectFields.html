        // 3. Select all columns for only [OrderDetails]
        Odb.SQL.Select(x => x.From(t0).LeftJoin(t1, t0.OrderID == t1.OrderID).Fields(t1) );

        // 4. Select all columns by excluding column [ShipperID] from [Orders]
