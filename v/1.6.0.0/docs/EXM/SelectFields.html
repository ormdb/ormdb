Following section demonstrates how select/exclude the fields to be read. 
		var th = new TableHeader();
		var tl = new TableLines();

		// selects all fields
		var sql = Odb.SQL.Select(x => x.From(th) );

		// selects only [Year] and [Id] fields from [TableLines]
		var sql = Odb.SQL.Select(x => 
						x.Fields(th.Year, th.Id)
						  .From(th) );

		// selects all fields for only [TableLines]
		var sql = Odb.SQL.Select(x => 
						x.Fields(tl)
						  .From(th)
						  .LeftJoin(tl, tl.Id == th.Id) );

		// selects all fields excluding field [Note]
		var sql = Odb.SQL.Select(x => 
						x.ExcludeFields(th.Note)
						  .From(th) );
	
		Dim th = new TableHeader()
		Dim tl = new TableLines()

		' selects all fields
		Dim sql = Odb.SQL.Select(Function(X) x.From(th) )

		' selects only [Year] and [Id] fields from [TableLines]
		Dim sql = Odb.SQL.Select(Function(X) _
						x.Fields(th.Year, th.Id) _
						  .From(th) )

		' selects all fields for only [TableLines]
		Dim sql = Odb.SQL.Select(Function(X) _
						x.Fields(tl) _
						  .From(th) _
						  .LeftJoin(tl, tl.Id = th.Id) )

		' selects all fields excluding field [Note]
		Dim sql = Odb.SQL.Select(Function(X) _
						x.ExcludeFields(th.Note) _
						  .From(th) )
