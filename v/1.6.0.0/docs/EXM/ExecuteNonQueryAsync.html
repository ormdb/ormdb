Following section demonstrates how to execute a "NonQuery" command as asynchronously by creating a task. 
		var th = new TableHeader();
		
		var sql = Odb.SQL.Delete(x => x.From(th).Where(th.Id == 12345M) );

		var task = Task<int>.Factory.StartNew(() => sql.ExecuteNonQuery(cn) );
		var affectedRows = task.Result;
	
		Dim th = new TableHeader()
		
		Dim sql = Odb.SQL.Delete(Function(x) x.From(th).Where(th.Id = 12345D) )
		
		Dim task = Task(Of Integer).Factory.StartNew(Function() sql.ExecuteNonQuery(cn))
		Dim affectedRows = task.Result
