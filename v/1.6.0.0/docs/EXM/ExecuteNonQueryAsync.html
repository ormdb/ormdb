Following section demonstrates how to execute a "NonQuery" command as asynchronously by creating a task. 
		var t0 = new Orders();
		var sql = Odb.SQL.Delete(x => x.From(t0).Where(t0.OrderID == 10248) );

		var task = Task<int>.Factory.StartNew(() => sql.ExecuteNonQuery(connection) );
		var affectedRows = task.Result;
