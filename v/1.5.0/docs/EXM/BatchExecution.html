Following section demonstrates how to execute multiple sql commands in one batch-query.

			for (n = 1; n <= 100; n++) {
				var row = _insert.AddRow(h.NewModel());
					row.Year = 2017;
					row.Id = 9999;
					row.Note = String.Concat("Note -", n.toString());
					row.DocDate = DateTime.Now();
					row.IsActive = true;
			}

			//-- update
			var row = h.NewModel();
				row.Note = "new note";
				row.DocDate = DateTime.Now();
			sql.Update(h, h.Id == 5).Set(row);

			'//-- delete
			.Delete(h)

			Dim _insert = .Insert(true);

					row.Year = 2017
					row.Id = n
					row.Note = String.Concat("Note -", n.toString())
					row.DocDate = DateTime.Now()

			Dim row = h.NewModel()
