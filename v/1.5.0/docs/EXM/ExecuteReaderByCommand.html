Following section demonstrates how to retreive the records by linq as dictionary. 

        var cmd = cn.CreateCommand("SELECT field1, field2 FROM table1 WHERE field1<>@param1;",
                                   new { param1 = 0});

        var result = cmd.ExecuteReader(cn).ToDictionary(
                                               x => x["field1"], 
                                               x => x["field2"]
                                           );
 
   
        Dim cmd = cn.CreateCommand("SELECT field1, field2 FROM table1 WHERE field1<>@param1;",
                                   new { .param1 = 0});

        Dim result = cmd.ExecuteReader(cn).ToDictionary(
                                              Function(x) x("field1"), 
                                              Function(x) x("field2")
                                          )
