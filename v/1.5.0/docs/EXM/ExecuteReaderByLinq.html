Following section demonstrates how to read record(s) with ExecuteReader, using Linq. 
        // Query is being generated...
        var cmd = SQL.X.Select().From(h);

        // Records are being read and converted to dictionary by linq...
        var result = cmd.ExecuteReader(cn).Select(x => new {
                Year = x(h.Year).ToInt32(),
                Id = x(h.Id).ToString(),
                Note = x(h.Note).ToString(),
                DocDate = x(h.DocDate).ToDateTime(),
                IsActive = x(h.IsActive).ToBoolean()
            }).ToList();
 
   
        ' Query is being generated...
        Dim cmd = SQL.X.Select().From(h);

        ' Records are being read and converted to dictionary by linq...
        Dim result = cmd.ExecuteReader(cn).Select(Function(x) New With {
                .Year = x(h.Year).ToInt32(),
                .Id = x(h.Id).ToString(),
                .Note = x(h.Note).ToString(),
                .DocDate = x(h.DocDate).ToDateTime(),
                .IsActive = x(h.IsActive).ToBoolean()
            }).ToList()
