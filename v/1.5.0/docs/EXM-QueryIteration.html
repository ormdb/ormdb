Following example iterates the records for two columns. 

		var sql = SQL.X.Select(h.Note, h.DocDate).From(h);

		foreach (var r in sql.ExecuteReader(cn)) {
			// read by column-name.
			Debug.WriteLine("{0}, {1}", r("Note").ToString(), 
										  r("DocDate").ToString());

			// read by column.
			Debug.WriteLine("{0}, {1}", r(h.Note).ToString(), 
										  r(h.DocDate).ToString());

			// read by column-index.
			Debug.WriteLine("{0}, {1}", r(0).ToString(),
										  r(1).ToString());
		}
