		// new instance of a table from definition
		var h = new TableHeader();
		
		// copies records in 2017 into same table as 2018
		var sql = SQL.X.Select(h.Const(2018).AS(h.Year), 
							   h.Id,
							   h.Note,
							   h.DocDate).From(h, h.Year == 2017);
		
		
		// query is being executed
		SQL.X.InsertInto(h, sql).ExecuteNonQuery(cn);
