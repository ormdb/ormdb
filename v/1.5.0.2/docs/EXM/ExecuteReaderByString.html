Following section demonstrates how to read record(s) with ExecuteReader, using SQL String. 
        // SQL statement is being prepared
        var sql = "SELECT field1, field2 FROM table1 AS h WHERE field3 = 2017";

        // Records are being read and converted to dictionary by linq...
        var dic = cn.ExecuteReader(sql).ToDictionary(
                                       x => x["field1"], 
                                       x => x["field2"]
                                   );
	
        ' SQL statement is being prepared
        Dim sql = "SELECT field1, field2 FROM table1 AS h WHERE field3 = 2017"

        ' Records are being read and converted to dictionary by linq...
        Dim dic = cn.ExecuteReader(sql).ToDictionary(
                                      Function(x) x("field1"), 
                                      Function(x) x("field2")
                                  )
